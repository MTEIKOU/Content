  int tonearg[]={
    262,294,330,349,392,440,494,523};
  int n;

  void setup() {
    Serial.begin(9600);
  }

  void loop() {

    int a = analogRead(0);
    Serial.println(a);

    if(a>1020){
      noTone(9);
    }
    else{

      for(n=0;n<8;n++){
        if(level[n]-3a){
          tone(9,tonearg[n],100);
        }
      }
    }
  }
